#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    long long a[(int)2e5+1];
    cin>>n;
    set<long long> s;
    for(int i = 0; i < n; ++i ){
        cin>>a[i];
    }
    set<long long>::iterator it;
    long long temp = (long long)100000000000000000;
    for(int i = n - 1; i >= 0; --i){
        s.insert(a[i]);
        it = s.find(a[i]);
        if(it != s.begin()){
            --it;
            temp = min(temp,(long long)a[i] - (*it));
        }
    }
    cout<<temp<<endl;
    return 0;
}
